#!/bin/tcsh -xef

#created June 2018 (Catherine Cunningham)

# runs freesurfer segmentation/parcellation on T1-weighted volume
# ROIs can be used later for fiber tracking or other dti analysis

set Sublist = (S01)

foreach subj ($Sublist)

set path_P_ss = /Volumes/shares/BNU/cnsraid/schintu/data_analysis/Exp2_DTI/Subjects_proc/{$subj}

set ss_id = $path_P_ss:t

##############################

echo
echo ////// {$subj} recon-all //////
echo

# FS function

recon-all                                     \
    -all                                      \
    -sd       $path_P_ss                      \
    -subjid   anat_02                         \
    -i        $path_P_ss/anat_01/t1w.nii.gz

##############################

echo
echo ////// {$subj} suma make spec fs ////// 
echo

# AFNI-SUMA function to convert FS output

@SUMA_Make_Spec_FS                            \
    -NIFTI                                    \
    -fspath $path_P_ss/anat_02                \
    -sid    $ss_id

end #subj
